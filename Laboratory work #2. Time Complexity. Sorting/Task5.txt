1561. Maximum Number of Coins You Can Get
LeetCode: https://leetcode.com/problems/maximum-number-of-coins-you-can-get/

class Solution {
    public int maxCoins(int[] piles) {
        Arrays.sort(piles);
        
        int index = piles.length - 2;
        int sum = 0;
        
        while(true) {
            sum += piles[index];
            
            if(piles.length / 3 == index)
               return sum;
            
            index -= 2;
        }
    }
}
